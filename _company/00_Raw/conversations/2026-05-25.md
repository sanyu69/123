class StabilityMetricPayload:
    def __init__(self, user_id: str, stability_score: float, test_group: str, timestamp: int):
        self.user_id = user_id
        self.stability_score = stability_score
        self.test_group = test_group
        self.timestamp = timestamp

    def to_dict(self) -> Dict[str, Any]:
        return {
            "user_id": self.user_id,
            "stability_score": self.stability_score,
            "test_group": self.test_group,
            "timestamp": self.timestamp
        }

def calculate_stability(input_data: Dict[str, Any]) -> float:
    """
    입력 데이터를 기반으로 시스템 안정성 지표($R_{Stability}$)를 계산하는 로직 (가상 로직)
    실제로는 데이터 파이프라인에서 가져온 값을 사용하도록 설계합니다.
    """
    # 이 부분은 실제 $R_{Stability}$ 계산 로직에 따라 수정되어야 합니다.
    if 'stability_data' in input_data:
        return input_data['stability_data'].get('R_Stability', 0.5) # 예시 값 반환
    return 0.0

def record_user_reaction(user_id: str, group: str, reaction_value: float) -> Dict[str, Any]:
    """
    사용자의 반응을 기록하고 데이터베이스에 저장하는 함수 (가상 API 통합)
    """
    timestamp = int(time.time())
    # 실제 DB 저장 로직은 다음 단계에서 구현될 API 호출로 대체됨
    record = StabilityMetricPayload(
        user_id=user_id,
        stability_score=reaction_value,
        test_group=group,
        timestamp=timestamp
    )
    print(f"--- [DATA RECORDED] User {user_id} ({group}): Score={reaction_value}, Time={timestamp}")
    return record.to_dict()

def run_ab_test_simulation(num_users: int = 100, group_a_ratio: float = 0.5) -> Dict[str, Any]:
    """
    A/B 테스트 시뮬레이션을 실행하여 데이터 확보 단계를 시작합니다.
    """
    results = {
        "group_A": {"count": 0, "total_score": 0.0},
        "group_B": {"count": 0, "total_score": 0.0}
    }

    for i in range(num_users):
        # 그룹 할당 (50/50 분할)
        if i < int(num_users * group_a_ratio):
            group = "A"
        else:
            group = "B"

        # 가상의 반응 값 생성 (실제로는 API 호출로 대체됨)
        reaction = 0.5 + (i % 100) / 200 # 0.5 ~ 1.0 사이의 랜덤 값

        # 데이터 기록 시뮬레이션
        recorded_data = record_user_reaction(f"user_{i:03d}", group, reaction)

        if group == "A":
            results["group_A"]["count"] += 1
            results["group_A"]["total_score"] += reaction
        else:
            results["group_B"]["count"] += 1
            results["group_B"]["total_score"] += reaction

    # 최종 집계
    avg_a = results["group_A"]["total_score"] / results["group_A"]["count"] if results["group_A"]["count"] > 0 else 0
    avg_b = results["group_B"]["total_score"] / results["group_B"]["count"] if results["group_B"]["count"] > 0 else 0

    return {
        "simulation_results": {
            "Group A": {"Count": results["group_A"]["count"], "Avg R_Stability": round(avg_a, 4)},
            "Group B": {"Count": results["group_B"]["count"], "Avg R_Stability": round(avg_b, 4)}
        },
        "total_users": num_users
    }

if __name__ == "__main__":

import time
import json
from typing import Dict, Any

# StabilityMetricPayload의 구조를 가정하고 정의합니다. (실제 데이터 구조는 API 통합 단계에서 확정될 예정)
class StabilityMetricPayload:
    def __init__(self, user_id: str, stability_score: float, test_group: str, timestamp: int):
        self.user_id = user_id
        self.stability_score = stability_score
        self.test_group = test_group
        self.timestamp = timestamp

    def to_dict(self) -> Dict[str, Any]:
        return {
            "user_id": self.user_id,
            "stability_score": self.stability_score,
            "test_group": self.test_group,
            "timestamp": self.timestamp
        }

def calculate_stability(input_data: Dict[str, Any]) -> float:
    """
    입력 데이터를 기반으로 시스템 안정성 지표($R_{Stability}$)를 계산하는 로직 (가상 로직)
    실제로는 데이터 파이프라인에서 가져온 값을 사용하도록 설계합니다.
    """
    # 이 부분은 실제 $R_{Stability}$ 계산 로직에 따라 수정되어야 합니다.
    if 'stability_data' in input_data:
        return input_data['stability_data'].get('R_Stability', 0.5) # 예시 값 반환
    return 0.0

def record_user_reaction(user_id: str, group: str, reaction_value: float) -> Dict[str, Any]:
    """
    사용자의 반응을 기록하고 데이터베이스에 저장하는 함수 (가상 API 통합)
    """
    timestamp = int(time.time())
    # 실제 DB 저장 로직은 다음 단계에서 구현될 API 호출로 대체됨
    record = StabilityMetricPayload(
        user_id=user_id,
        stability_score=reaction_value,
        test_group=group,
        timestamp=timestamp
    )
    print(f"--- [DATA RECORDED] User {user_id} ({group}): Score={reaction_value}, Time={timestamp}")
    return record.to_dict()

def run_ab_test_simulation(num_users: int = 100, group_a_ratio: float = 0.5) -> Dict[str, Any]:
    """
    A/B 테스트 시뮬레이션을 실행하여 데이터 확보 단계를 시작합니다.
    """
    results = {
        "group_A": {"count": 0, "total_score": 0.0},
        "group_B": {"count": 0, "total_score": 0.0}
    }

    for i in range(num_users):
        # 그룹 할당 (50/50 분할)
        if i < int(num_users * group_a_ratio):
            group = "A"
        else:
            group = "B"

        # 가상의 반응 값 생성 (실제로는 API 호출로 대체됨)
        reaction = 0.5 + (i % 100) / 200 # 0.5 ~ 1.0 사이의 랜덤 값

        # 데이터 기록 시뮬레이션
        recorded_data = record_user_reaction(f"user_{i:03d}", group, reaction)

        if group == "A":
            results["group_A"]["count"] += 1
            results["group_A"]["total_score"] += reaction
        else:
            results["group_B"]["count"] += 1
            results["group_B"]["total_score"] += reaction

    # 최종 집계
    avg_a = results["group_A"]["total_score"] / results["group_A"]["count"] if results["group_A"]["count"] > 0 else 0
    avg_b = results["group_B"]["total_score"] / results["group_B"]["count"] if results["group_B"]["count"] > 0 else 0

    return {
        "simulation_results": {
            "Group A": {"Count": results["group_A"]["count"], "Avg R_Stability": round(avg_a, 4)},
            "Group B": {"Count": results["group_B"]["count"], "Avg R_Stability": round(avg_b, 4)}
        },
        "total_users": num_users
    }

if __name__ == "__main__":

### 🎨 Designer를 위한 시각 및 톤앤매너 가이드라인

**1. 핵심 컨셉:** **Neon Stability (네온 안정성)**
*   **핵심 메시지:** '안정성($R_{Stability}$) = 프리미엄 가치'
*   **시각적 톤:** 사이버펑크적이면서도 매우 정제되고 고급스러운 네온(Neon) 라이팅을 사용합니다. 어둡고 깊은 배경에, 데이터 흐름과 안정성을 상징하는 밝고 선명한 네온 라인이 대비를 이룹니다.
*   **VIP vs Pro 시각 차별화:**
    *   **VIP (깊이):** 미니멀하고 광활한 공간감(Depth)을 강조하며, 은은하게 빛나는 깊은 파란색/보라색 계열의 네온 사용. '여유로움'과 '깊이 있는 통제력'을 표현.

## ✨ 시각 컨셉 제안: Neon Stability - Depth vs Density

핵심 목표는 **'안정성'**이라는 추상적인 개념을 **'깊이(Depth)'**와 **'밀도(Density)'**라는 구체적인 시각적 차이로 명확하게 분리하여 사용자에게 프리미엄 경험의 질적 차이를 즉시 인지시키는 것입니다.

### 💡 핵심 디자인 원칙 (Core Principles)

1.  **컬러 팔레트:** 'Neon Stability'를 유지하되, 티어별로 색상의 채도와 명도를 조절하여 차별화합니다.
    *   **기본 네온:** `#00FFFF` (시안/청록), `#FF00FF` (마젠타/핑크)
    *   **Depth (VIP):** 부드러운 그라데이션, 깊은 그림자(Shadow), 낮은 채도. (신뢰감, 몰입감)
    *   **Density (Pro):** 날카로운 라인, 고대비, 밀집된 그리드 패턴, 높은 채도. (정보력, 강력함)
2.  **텍스처:** 깊이는 유기적인 흐름(Flow), 밀도는 기하학적 구조(Grid/Matrix)를 사용하여 표현합니다.
3.  **레이아웃:** 썸네일은 좌우 분할 또는 대비되는 질감의 병치를 통해 차이를 극대화합니다.

---

### 🖼️ 썸네일 컨셉 3안 (Thumbnail Concepts)

#### 🌟 컨셉 1: 이중 노출 (The Dual Exposure) - **가장 추천**
*   **테마:** 대비(Contrast)와 깊이의 시각적 표현.
*   **레이아웃:** 화면을 수직으로 정확히 반으로 분할합니다.
    *   **좌측 (VIP/Depth):** 어둡고 깊은 네온 블루/퍼플 배경 위에 부드럽게 빛나는 메인 캐릭터나 게임 아트의 **'깊은 그림자(Deep Shadow)'** 효과를 강조. 마치 심해처럼 몰입감 있는 분위기 연출.
    *   **우측 (Pro/Density):** 밝고 날카로운 네온 그린/마젠타 배경 위에 복잡한 데이터 그래프, 선형 그리드, 정보가 밀집된 **'네온 그리드(Neon Grid)'** 패턴을 강조.
*   **텍스트 오버레이:** 중앙에 "Depth vs Density" 타이포그래피를 명확히 배치하고, VIP/Pro 라벨을 각 영역에 겹쳐 표시합니다.
*   **느낌:** 고급스럽고, 차별화가 즉각적으로 인식되며, 프리미엄 등급의 느낌이 강함.

#### 🌟 컨셉 2: 흐름과 구조 (Flow & Structure) - **구조적 강조형**
*   **테마:** 안정성의 물리적 구현.
*   **레이아웃:** 전체 화면을 하나의 거대한 네온 프레임으로 감쌉니다.
    *   **VIP 영역:** 부드러운 빛의 흐름(Flowing Light)이 화면을 채우며, 마치 액체처럼 유연하게 움직이는 듯한 **'안정적인 안정성'**을 표현합니다. (물결 또는 안개 효과 사용)
    *   **Pro 영역:** 정교하고 날카로운 기하학적 라인과 데이터 포인트가 촘촘하게 박혀있는 **'밀도 높은 구조(Dense Structure)'**를 강조합니다. 마치 고해상도 회로 기판처럼 보입니다.
*   **텍스트 오버레이:** 제목은 명료하게 "VIP: Depth / Pro: Density"로 제시하여 각 요소의 정의를 명확히 합니다.
*   **느낌:** 과학적이고, 시스템적인 안정성이 강조되며, 기술적 프리미엄이 느껴짐.

#### 🌟 컨셉 3: 네온 스펙트럼 (Neon Spectrum) - **색상 대비 강조형**
*   **테마:** 색채를 통한 감성적 차별화.
*   **레이아웃:** 배경은 단색(예: 딥 블랙 또는 미드나잇 블루). 두 영역은 색상만으로 구분합니다.
    *   **VIP (Depth):** 차분하고 깊은 **네온 퍼플/블루 그라데이션**. 빛이 부드럽게 번지는 느낌을 주어 '깊이'를 시각화합니다.
    *   **Pro (Density):** 채도가 매우 높은 **핫 네온 그린/레드**의 선과 점으로 이루어진 밀집된 패턴을 사용합니다. 이는 정보가 꽉 차 있다는 인상을 줍니다.
*   **텍스트 오버레이:** 색상 자체를 타이포그래피처럼 활용하여 'Depth'와 'Density'라는 키워드를 각 영역에 배치합니다.
*   **느낌:** 강렬하고 감각적이며, 네온 아트의 미학을 극대화하여 시선을 사로잡음.
