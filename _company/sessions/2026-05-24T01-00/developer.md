import json
from typing import Dict, Any

# $R_{Stability}$ 데이터 모델 및 VIP 지표 매핑 규칙을 참조합니다. (API_Specification.md 기반)
# 이 파일은 시스템 안정성 데이터를 기반으로 VIP 안정성 점수($S_{VIP}$)를 산출하는 핵심 로직을 담습니다.

def calculate_vip_stability(stability_data: Dict[str, float], vip_mapping: Dict[str, Any]) -> float:
    """
    시스템 안정성 데이터와 VIP 매핑 규칙을 기반으로 VIP 안정성 점수($S_{VIP}$)를 계산합니다.
    $R_{Stability}$ 데이터의 무결성을 보장하는 것이 최우선입니다.

    Args:
        stability_data (Dict[str, float]): 시스템 안정성 지표 ($R_{Stability}$) 데이터.
        vip_mapping (Dict[str, Any]): VIP 티어별 안정성 가중치 및 임계값 매핑 규칙.

    Returns:
        float: 계산된 VIP 안정성 점수 ($S_{VIP}$).
    """
    if not stability_data or not vip_mapping:
        # 데이터가 없거나 매핑 규칙이 없으면 0으로 처리하여 시스템을 보호합니다.
        print("Warning: Missing stability data or VIP mapping rules.")
        return 0.0

    # 핵심 안정성 지표($R_{Stability}$)를 기반으로 가중치 적용
    stability_score = stability_data.get('overall_stability', 0.0)
    vip_level = stability_data.get('current_tier', 'Basic') # 현재 VIP 등급

    # VIP 티어별 안정성 가중치를 가져옵니다.
    if vip_level in vip_mapping:
        weight = vip_mapping[vip_level].get('weight', 1.0)
        threshold = vip_mapping[vip_level].get('threshold', 50.0)

        # 안정성 점수를 가중치와 임계값을 적용하여 $S_{VIP}$를 산출합니다.
        if stability_score >= threshold:
            vip_score = stability_score * weight
        else:
            # 임계값 미달 시 페널티 또는 낮은 점수 부여
            vip_score = stability_score * 0.5

        return round(vip_score, 2)
    else:
        # 매핑 규칙이 없는 경우 기본 안정성 점수를 반환합니다.
        return round(stability_score, 2)


def process_vip_request(input_data: Dict[str, Any]) -> Dict[str, Any]:
    """
    API 요청을 받아 $S_{VIP}$를 계산하고 결과를 반환하는 엔드포인트 로직.
    안정적인 데이터 흐름을 보장합니다.
    """
    try:
        stability_data = input_data.get('stability_metrics', {})
        
        if not stability_data:
            raise ValueError("Stability metrics are missing in the request.")

        # VIP 매핑 규칙은 설정 파일에서 로드되어야 하지만, 현재는 임시로 하드코딩된 구조를 가정합니다. (실제 구현 시 설정 파일 로딩 필요)
        vip_mapping = {
            "Basic": {"weight": 1.0, "threshold": 50.0},
            "Pro": {"weight": 1.2, "threshold": 75.0},
            "VIP": {"weight": 1.5, "threshold": 90.0}
        }

        vip_score = calculate_vip_stability(stability_data, vip_mapping)

        return {
            "status": "success",
            "message": "VIP Stability Score calculated successfully.",
            "vip_stability_score_svip": vip_score,
            "details": {
                "R_Stability": stability_data.get('overall_stability'),
                "CurrentTier": stability_data.get('current_tier')
            }
        }

    except ValueError as e:
        return {"status": "error", "message": str(e), "code": 400}
    except Exception as e:
        # 모든 예외 상황에서 안정성을 위해 에러를 명확히 반환합니다.
        print(f"Error during VIP processing: {e}")
        return {"status": "error", "message": "Internal server error during calculation.", "code": 500}

# 테스트용 실행 함수 (실제 API 통합 전에 로직 검증)
if __name__ == '__main__':
    # 예시 데이터
    example_stability = {
        'overall_stability': 92.5,  # R_Stability
        'current_tier': 'VIP',
        'latency_ms': 45.0
    }
    
    vip_rules = {
        "Basic": {"weight": 1.0, "threshold": 50.0},
        "Pro": {"weight": 1.2, "threshold": 75.0},
        "VIP": {"weight": 1.5, "threshold": 90.0}
    }

    result = process_vip_request(example_stability)
    print(json.dumps(result, indent=4))
