import json

# R_Stability 데이터 구조 정의
class StabilityMetrics:
    def __init__(self, stability_score: float, vip_threshold: float, pro_threshold: float):
        self.stability_score = stability_score  # 시스템 안정성 지표 (0.0 ~ 1.0)
        self.vip_threshold = vip_threshold      # VIP 전환 기준 임계값
        self.pro_threshold = pro_threshold      # Pro 전환 기준 임계값

    def calculate_tier(self, current_score: float) -> str:
        """현재 점수를 기반으로 티어를 계산합니다."""
        if current_score >= self.pro_threshold:
            return "Pro"
        elif current_score >= self.vip_threshold:
            return "VIP"
        else:
            return "Basic"
