 35	STABILITY_THRESHOLDS = {

 40	def calculate_action_recommendation(r_stability: float, s_success: float) -> Dict[str, Any]:
 41	    """
 42	    시스템 안정성($R_{Stability}$)과 수익화 목표($S_{Success}$)를 기반으로 행동 권고안을 계산합니다.
 43	
 44	    Args:
 45	        r_stability (float): 시스템 안정성 지표 (0.0 ~ 1.0)
 46	        s_success (float): 수익화 목표 달성 지표 (0.0 ~ 1.0)
 47	
 48	    Returns:
 49	        Dict[str, Any]: 행동 권고안 및 상태 정보
 50	    """
 51	    recommendation = {
 52	        "status": "NORMAL",
 53	        "recommendation": "No immediate action required.",
 54	        "reason": "System stability and success metrics are within acceptable bounds.",
 55	        "action_level": "LOW"
 56	    }
 57	
 58	    # 1. 안정성(R_Stability) 기반 판단
 59	    if r_stability <= STABILITY_THRESHOLDS["CRITICAL"]:
 60	        recommendation["status"] = "CRITICAL"
 61	        recommendation["action_level"] = "HIGH"
 62	        recommendation["recommendation"] = "Immediate stability review required. Check system logs for critical errors."
 63	        recommendation["reason"] = f"R_Stability ({r_stability:.2f}) is below the CRITICAL threshold ({STABILITY_THRESHOLDS['CRITICAL']})."
 64	
 65	    elif r_stability <= STABILITY_THRESHOLDS["WARNING"]:
 66	        recommendation["status"] = "WARNING"
 67	        recommendation["action_level"] = "MEDIUM"
 68	        recommendation["recommendation"] = "Monitor system performance closely. Investigate potential bottlenecks."
 69	        recommendation["reason"] = f"R_Stability ({r_stability:.2f}) is below the WARNING threshold ({STABILITY_THRESHOLDS['WARNING']})."
 70	
 71	    # 2. 수익화 목표(S_Success) 기반 판단 (안정성이 확보된 경우에만 적용)
 72	    elif s_success < 0.7 and r_stability > STABILITY_THRESHOLDS["WARNING"]:
 73	        recommendation["status"] = "STAGNANT"
 74	        recommendation["action_level"] = "MEDIUM"
 75	        recommendation["recommendation"] = "Focus on optimization strategies to increase $S_{Success}$. Stability is adequate for growth initiatives."
 76	        recommendation["reason"] = f"Growth focus needed. R_Stability ({r_stability:.2f}) is stable, but S_Success ({s_success:.2f}) needs attention."
 77	
 78	    elif s_success >= 0.9 and r_stability > STABILITY_THRESHOLDS["WARNING"]:
 79	        recommendation["status"] = "OPTIMAL"
 80	        recommendation["action_level"] = "LOW"
 81	        recommendation["recommendation"] = "System is operating optimally. Focus on maintenance."
 82	        recommendation["reason"] = f"System is stable and success metrics are high. $R_{Stability}$ ({r_stability:.2f}) and $S_{Success}$ ({s_success:.2f}) are excellent."
 83	
 84	    return recommendation

 86	def generate_daily_briefing(metrics: Dict[str, float]) -> Dict[str, Any]:
 87	    """
 88	    모든 에이전트 활동과 지표를 통합하여 최종 데일리 브리핑 템플릿을 생성합니다.
 89	    """
 90	    r_stability = metrics.get("R_Stability", 0.5)

 93	    # 핵심 행동 권고안 계산
 94	    action_result = calculate_action_recommendation(r_stability, s_success)
 95	
 96	    briefing = {
 97	        "Date": "2026-05-27", # 실제 실행 시 날짜 동적 변경 필요
 98	        "System_Stability_R_Stability": r_stability,
 99	        "Success_Metric_S_Success": s_success,
100	        "Action_Recommendation": action_result,
101	        "Summary": f"Overall Status: {action_result['status']}. Next Step: {action_result['recommendation']}",
102	        "Detailed_Analysis": (
103	            f"R_Stability: {r_stability:.2f} | S_Success: {s_success:.2f}\n"
104	            f"Threshold Check: CRITICAL={STABILITY_THRESHOLDS['CRITICAL']}, WARNING={STABILITY_THRESHOLDS['WARNING']}"
105	        )
106	    }
107	
108	    return briefing
